## Posterior means of the latent space positions, with the movements interpolated
library(animation)
library(vegan)
library(lattice)
library(permute)
library(RcppArmadillo)
library(inline)
library(MCMCpack)
library(MASS)

par(mar = c(2, 2, 2, 2) + 0.1)
for (tt in 1:(TT - 1)) {
    for (l in 1:length(L)) {
        plot(L[l] * t(EX[, tt, ]) + (1 - L[l]) * t(EX[, tt + 
            1, ]), xlab = "", ylab = "", xlim = lims, ylim = lims, 
            pch = 16, xaxt = "n", yaxt = "n", cex = 2)
        ani.pause()
    }
}
